compound assignment operators

The compound assignment operator hide one step of righthand side evaluation. This is where the code deviates with what’s in our mind.

let val = 0;

function inc1() {
 val++;
 console.log(val);
}

function inc2() {
 val += inc1();
 console.log(val);
}

inc2();
let val = 0;

function inc1() {
 val++;
 console.log(val);
}

function inc2() {
 val = val + inc1();
 console.log(val);
}

inc2();
